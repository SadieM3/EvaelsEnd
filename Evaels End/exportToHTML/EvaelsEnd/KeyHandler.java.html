10       @Override

12   
13       }
14   
15       @Override
16       public void keyPressed(KeyEvent e) {
17           int code = e.getKeyCode();
18           if (code == KeyEvent.VK_W){
19               upPressed = true;
20           }
21           if (code == KeyEvent.VK_A){
22               leftPressed = true;
23           }
24           if (code == KeyEvent.VK_S){
25               downPressed = true;
26           }
27           if (code == KeyEvent.VK_D){
28               rightPressed = true;
29           }
30       }
31   
32       @Override
33       public void keyReleased(KeyEvent e) {
34           int code = e.getKeyCode();
35           if (code == KeyEvent.VK_W){
36               upPressed = false;
37           }
38           if (code == KeyEvent.VK_A){
39               leftPressed = false;
40           }
41           if (code == KeyEvent.VK_S){
42               downPressed = false;
43           }
44           if (code == KeyEvent.VK_D){
45               rightPressed = false;
46           }
47       }
48   }
